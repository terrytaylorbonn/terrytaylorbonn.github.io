Stores messy human text in an append-only encrypted ledger, then uses AI to search, summarize, and generate new entries without requiring a rigid database schema.

The ledger stays immutable:
- Entry 001: original note
- Entry 002: correction to Entry 001
- Entry 003: update to Entry 001

Then the AI/agent produces the current “view”:
- current version = Entry 001 + Entry 002 + Entry 003

So nothing is edited in place. The system only appends.

# ledger.py

import json
import hashlib
from datetime import datetime, timezone
from pathlib import Path

LEDGER_FILE = Path("ledger.jsonl")
GENESIS_HASH = "0" * 64

def canonical_json(data):
    return json.dumps(data, sort_keys=True, separators=(",", ":"))

def sha256_text(text):
    return hashlib.sha256(text.encode("utf-8")).hexdigest()

def read_entries():
    if not LEDGER_FILE.exists():
        return []

    entries = []
    with LEDGER_FILE.open("r", encoding="utf-8") as f:
        for line in f:
            if line.strip():
                entries.append(json.loads(line))

    return entries

def append_event(event):
    entries = read_entries()

    entry_number = len(entries) + 1
    prev_hash = entries[-1]["hash"] if entries else GENESIS_HASH

    entry = {
        "entry": entry_number,
        "timestamp": datetime.now(timezone.utc).isoformat(),
        "event": event,
        "prev_hash": prev_hash,
    }

    entry["hash"] = sha256_text(canonical_json(entry))

    with LEDGER_FILE.open("a", encoding="utf-8") as f:
        f.write(json.dumps(entry) + "\n")

    return entry

def verify():
    entries = read_entries()

    prev_hash = GENESIS_HASH

    for entry in entries:
        stored_hash = entry["hash"]

        check_entry = dict(entry)
        del check_entry["hash"]

        expected_hash = sha256_text(canonical_json(check_entry))

        if stored_hash != expected_hash:
            return False, f"Bad hash at entry {entry['entry']}"

        if entry["prev_hash"] != prev_hash:
            return False, f"Bad prev_hash at entry {entry['entry']}"

        prev_hash = stored_hash

    return True, f"Ledger OK. Entries: {len(entries)}"

def show():
    entries = read_entries()

    for entry in entries:
        print(json.dumps(entry, indent=2))

if __name__ == "__main__":
    test_event = {
        "event_type": "test_event",
        "message": "Ledger test entry",
    }

    entry = append_event(test_event)
    print("Appended entry:")
    print(json.dumps(entry, indent=2))

    ok, msg = verify()
    print()
    print(msg)
